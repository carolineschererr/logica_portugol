algoritmo "Numero Primo"

var
C, N, ContDiv: Inteiro

inicio

C <- 1
ContDiv <- 0
Escreva("Digite um numero: ")
Leia(N)
Repita
	Escreva(C)
	Se (N % C = 0) entao
	ContDiv <- ContDiv+1
	FimSe
	C <- C+1
Ate (C > N)
Se(ContDiv > 2) entao
Escreval("O número não é primo.")
Senao
	EscrevaL("O número é primo.")
FimSe
